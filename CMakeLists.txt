  request STATIC
  include/request/request.h
  include/request/response.h
  include/request/response_p.h
  include/request/session.h
  include/request/session_p.h
  include/request/type.h

  response.cpp
  session.cpp
  curl_easy.h
  curl_multi.h
  curl_error.h
  type.cpp
  uri.cpp
  )

  request
  PUBLIC include
  PRIVATE include/request
  )

                                      fmt::fmt
                                      pegtl
                                      )
